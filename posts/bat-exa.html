-i/--ignore-case

fd

find replacement

fd text  #recursively
fd 'regular expression'

fd text /directory

fd --extension md  =  fd -e md
--hidden = -H

-x/--exec option runs an external command for each of the search results
-X/--exec-batch runs once for all results

--exclude = -E  #exclude directory
